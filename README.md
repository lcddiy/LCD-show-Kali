# How to install the driver
-------for 3.5inch GPIO--------------\
sudo rm -rf LCD-show-Kali\
git clone github.com/lcddiy/LCD-show-Kali.git\
chmod -R 755 LCD-show-Kali\
cd LCD-show-Kali\
sudo ./LCD35-show

-------for 3.5inch HDMI 480X320--------------\
sudo rm -rf LCD-show-Kali\
git clone github.com/lcddiy/LCD-show-Kali.git\
chmod -R 755 LCD-show-Kali\
cd LCD-show-Kali\
sudo ./LCD35-HDMI-480x320-show

-------for 3.5inch HDMI 600X400--------------\
sudo rm -rf LCD-show-Kali\
git clone github.com/lcddiy/LCD-show-Kali.git\
